Init Containers

initContainers:
  waitForService:
    enabled: true
    serviceName: kubernetes.default.svc.cluster.local
  download:
    enabled: true
    url: https://example.com
    fileName: index.html

Verification

kubectl get pods -n monitoring-app -w
kubectl logs <pod> -c init-download
kubectl exec <pod> -- cat /init-data/index.html
